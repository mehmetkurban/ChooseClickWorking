$(document).ready(function(){

    $('.container').hide(0);
       $('#deneme').hide(0);
          $('#deneme2').hide(0);
    
    $("a").click(function(){
        $(".container").hide();
    });
    
    $("p").on({
    mouseenter: function(){
        $(this).css("background-color", "lightgray");
    }, 
    mouseleave: function(){
        $(this).css("background-color", "lightblue");
    }, 
    click: function(){
        $(this).css("background-color", "yellow");
    } 
    
    });
    
    
    $("button").on({
        click: function(){             
            $(".container").fadeIn(0);
            $("#deneme").fadeIn("slow");
            $("#deneme2").fadeIn(3000);
        },
        mouseenter: function(){
            $(".container").fadeOut(0);
            $("#deneme").fadeOut("slow");
            $("#deneme2").fadeOut(3000);
        }
    });
    
    $("div").hide();
    
    
    
});
